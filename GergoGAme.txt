-- Scripts:

local function BDWNC_fake_script() -- GainMuscle.LocalScript 
	local script = Instance.new('LocalScript', GainMuscle)

	script.Parent.MouseButton1Click:Connect(function()
		_G.LULU = true
		while _G.LULU do
			wait()
			game:GetService("ReplicatedStorage").GetMuscle:FireServer()
		end
	end)

	script.Parent.MouseButton1Click:Connect(function()
		script.Parent.Visible = false
		script.Parent.Parent.GainMuscleOFF.Visible = true
	end)
end
coroutine.wrap(BDWNC_fake_script)()
